text

// Teks biasa, support link preview otomatis.

text.js
await sock.sendMessage(m.chat, {
  text: 'Halo bro!'
}, { quoted: m })

// matiin link preview
await sock.sendMessage(m.chat, {
  text: 'https://example.com',
  linkPreview: null
})

image

// Gambar dari URL, buffer, atau path file lokal.

image.js
await sock.sendMessage(m.chat, {
  image: { url: 'https://example.com/img.jpg' },
  caption: 'Caption opsional'
})

// dari buffer
await sock.sendMessage(m.chat, {
  image: buffer,
  caption: 'dari buffer',
  viewOnce: true  // view once
})

video.js

react

// React emoji ke pesan. text: '' = hapus reaksi.

react.js
await sock.sendMessage(m.chat, {
  react: { text: '🔥', key: m.key }
})

// hapus reaksi
await sock.sendMessage(m.chat, {
  react: { text: '', key: m.key }
})

})

payment new

// Request pembayaran ke user.

payment.js
await sock.sendMessage(m.chat, {
  payment: {
    amount: 50000,
    currency: 'IDR',
    note: 'Bayar dong bro',
    from: '628xxx@s.whatsapp.net',
    expiry: 0
  }
}, { quoted: m })
field
type
keterangan
amount
number
Nominal (rupiah untuk IDR)
currency
string
ISO 4217 — default: 'IDR'
note opt
string
Catatan payment
from opt
string
JID yang diminta bayar
expiry opt
number
Unix timestamp, default: 0 (no expiry)
